            // Listen for real-time updates  
            this.listRef.on('value', (snapshot) => {  
                const data = snapshot.val();  
                if (data && data.items) {  
                    this.items = Object.values(data.items).sort((a, b) => b.order - a.order);  
                } else {  
                    this.items = [];

        getListId() {  
            // Get list ID from URL or create new one  
            const urlParams = new URLSearchParams(window.location.search);  
            let listId = urlParams.get('list');  
              
            if (!listId) {  
                listId = localStorage.getItem('currentListId');  
            }  
              
            if (!listId) {  
                listId = 'list_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);  
                localStorage.setItem('currentListId', listId);  
            }  
              
            return listId;  
        }  

        addItem() {  
            const input = document.getElementById('newItem');  
            const text = input.value.trim();  
              
            if (text) {  
                const itemId = 'item_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);  
                const item = {  
                    id: itemId,  
                    text: text,  
                    completed: false,  
                    createdAt: new Date().toISOString(),  
                    order: Date.now()  
                };  
                  
                // Save to Firebase  
                this.listRef.child('items').child(itemId).set(item)  
                    .then(() => {  
                        input.value = '';  
                        this.showSyncStatus('syncing');  
                    })  
                    .catch((error) => {  
                        console.error('Error adding item:', error);  
                        this.showSyncStatus('error');  
                    });  
            }  
        }  

        toggleItem(id) {  
            this.showSyncStatus('syncing');  
            this.listRef.child('items').child(id).child('completed').transaction((completed) => {  
                return !completed;  
            }).catch((error) => {  
                console.error('Error toggling item:', error);  
                this.showSyncStatus('error');  
            });  
        }  

        editItem(id, newText) {  
            if (newText.trim()) {  
                this.showSyncStatus('syncing');  
                this.listRef.child('items').child(id).child('text').set(newText.trim())  
                    .catch((error) => {  
                        console.error('Error editing item:', error);  
                        this.showSyncStatus('error');  
                    });  
            }  
        }  

        deleteItem(id) {  
            const itemElement = document.querySelector(`[data-id="${id}"]`);  
            if (itemElement) {  
                itemElement.classList.add('fade-out');  
                setTimeout(() => {  
                    this.showSyncStatus('syncing');  
                    this.listRef.child('items').child(id).remove()  
                        .catch((error) => {  
                            console.error('Error deleting item:', error);  
                            this.showSyncStatus('error');  
                        });  
                }, 300);  
            }  
        }  

        clearCompleted() {  
            const completedItems = document.querySelectorAll('.completed-item');  
            completedItems.forEach(item => item.classList.add('fade-out'));  
              
            setTimeout(() => {  
                this.showSyncStatus('syncing');  
                const updates = {};  
                this.items.forEach(item => {  
                    if (item.completed) {  
                        updates[`items/${item.id}`] = null;  
                    }  
                });  
                  
                this.listRef.update(updates)  
                    .catch((error) => {  
                        console.error('Error clearing completed items:', error);  
                        this.showSyncStatus('error');  
                    });  
            }, 300);  
        }  

        toggleAll() {  
            const allCompleted = this.items.every(item => item.completed);  
            const newCompletedState = !allCompleted;  
              
            this.showSyncStatus('syncing');  
            const updates = {};  
            this.items.forEach(item => {  
                updates[`items/${item.id}/completed`] = newCompletedState;  
            });  
              
            this.listRef.update(updates)  
                .catch((error) => {  
                    console.error('Error toggling all items:', error);  
                    this.showSyncStatus('error');  
                });  
        }  

        moveItem(fromIndex, toIndex) {  
            const item = this.items[fromIndex];  
            const targetItem = this.items[toIndex];  
              
            // Calculate new order value  
            let newOrder;  
            if (toIndex === 0) {  
                newOrder = (this.items[0]?.order || 0) + 1000;  
            } else if (toIndex === this.items.length - 1) {  
                newOrder = (this.items[this.items.length - 1]?.order || 0) - 1000;  
            } else {  
                const prevOrder = this.items[toIndex - 1]?.order || 0;  
                const nextOrder = this.items[toIndex + 1]?.order || 0;  
                newOrder = (prevOrder + nextOrder) / 2;  
            }  
              
            this.showSyncStatus('syncing');  
            this.listRef.child('items').child(item.id).child('order').set(newOrder)

                    console.error('Error moving item:', error);  
                    this.showSyncStatus('error');  
                });  
        }

        addDragAndDrop() {  
            const items = document.querySelectorAll('[draggable="true"]');  
              
            items.forEach(item => {  
                item.addEventListener('dragstart', (e) => {  
                    this.draggedItem = item;  
                    item.classList.add('dragging');  
                    e.dataTransfer.effectAllowed = 'move';  
                });  

                item.addEventListener('dragend', () => {  
                    item.classList.remove('dragging');  
                    this.draggedItem = null;  
                    // Remove all drag-over classes  
                    items.forEach(i => i.classList.remove('drag-over'));  
                });  

                item.addEventListener('dragover', (e) => {  
                    e.preventDefault();  
                    e.dataTransfer.dropEffect = 'move';  
                      
                    if (this.draggedItem && this.draggedItem !== item) {  
                        item.classList.add('drag-over');  
                    }  
                });  

                item.addEventListener('dragleave', () => {  
                    item.classList.remove('drag-over');  
                });  

                item.addEventListener('drop', (e) => {  
                    e.preventDefault();  
                    item.classList.remove('drag-over');  
                      
                    if (this.draggedItem && this.draggedItem !== item) {  
                        const fromIndex = parseInt(this.draggedItem.dataset.index);  
                        const toIndex = parseInt(item.dataset.index);  
                        this.moveItem(fromIndex, toIndex);  
                    }  
                });  
            });  
        }
